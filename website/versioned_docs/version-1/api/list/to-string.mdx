```scss
@use '@sass-fairy/list';

@debug list.to-string([10px 20px 'a']);
  // 10px 20px a

@debug list.to-string([10px, 20px, 'a']);
  // 10px, 20px, a

@debug list.to-string([10px 20px (a: 1, 2: b)])
  // '10px 20px {a: 1; 2: b}'
```
