```scss
@use '@sass-fairy/list';

$list: 'ant', 'bison', 'camel', 'bison', 'duck', 'elephant';

@debug list.slice($list, 3);
  // 'camel', 'bison', 'duck', 'elephant'

@debug list.slice($list, 3, 5);
  // 'camel', 'bison', 'duck'

@debug list.slice($list, -3);
  // 'bison', 'duck', 'elephant'
```
