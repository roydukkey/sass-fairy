```scss
@use '@sass-fairy/list';

// 1. Flatten to the default depth
$list: [0, 1, 2, [3, [4]]];

@debug list.flat($list);
  // [0, 1, 2, 3, [4]]

// 2. Flatten to a custom depth
$list: [0, 1, 2, [3, 4]];

@debug list.flat($list, 2);
  // [0, 1, 2, 3, 4]
```
