@use '@sass-fairy/list';

  // 'camel', 'bison', 'duck', 'elephant'

@debug list.slice($list, 3, 5);
  // 'camel', 'bison', 'duck'

@debug list.slice($list, -3);
  // 'bison', 'duck', 'elephant'
